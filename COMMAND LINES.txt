- AT LAUNCH: ACCESS ROUTE AND OPEN HEROKU
cd node-js-getting-started & heroku login

- OPEN DATABASE TO DO QUERIES
heroku pg:psql
\q para cerrar
\dt show tables
drop table characters; drop table chapters; drop table users; drop table environment;
create table users (name varchar(255) not null unique, pass varchar(255) not null, primary key (name, pass));
create table chapters (name varchar(255) not null, tale int not null, chapter int not null, primary key (name, tale, chapter), CONSTRAINT hasChapter FOREIGN KEY(name) REFERENCES users(name));
create table characters (name varchar(255) not null, tale int not null, character varchar(255) not null, gender varchar(1) not null, color varchar(3) not null, primary key (name, tale, character, gender), CONSTRAINT hasCharacter FOREIGN KEY(name) REFERENCES users(name));
create table environment (name varchar(255) not null, xPlayer int not null, yPlayer int not null, dirPlayer int not null, primary key (name), CONSTRAINT hasCharacter FOREIGN KEY(name) REFERENCES users(name));
