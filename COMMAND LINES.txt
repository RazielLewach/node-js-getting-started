- AT LAUNCH: ACCESS ROUTE AND OPEN HEROKU
cd node-js-getting-started & heroku login

- OPEN DATABASE TO DO QUERIES
heroku pg:psql

drop table enemies01; drop table player01; drop table entity01; drop table characters; drop table chapters; drop table users;
create table users (name varchar(255) not null unique, pass varchar(255) not null, taleplaying int not null, chapterplaying int not null, primary key (name, pass));
create table chapters (name varchar(255) not null, tale int not null, chapter int not null, primary key (name, tale, chapter), CONSTRAINT hasChapter FOREIGN KEY(name) REFERENCES users(name));
create table characters (name varchar(255) not null, tale int not null, character varchar(255) not null, gender varchar(1) not null, color varchar(3) not null, primary key (name, tale, character, gender), CONSTRAINT hasCharacter FOREIGN KEY(name) REFERENCES users(name));
create table entity01 (entity int not null, fuerzaEntity int not null, resistenciaEntity int not null, precisionEntity int not null, reflejosEntity int not null, percepcionEntity int not null, camuflajeEntity int not null, inteligenciaEntity int not null, voluntadEntity int not null, heridasCabezaEntity int not null, heridasCuerpoEntity int not null, heridasBrazosEntity int not null, heridasPiernasEntity int not null, primary key (entity));
create table player01 (name varchar(255) not null, entity int not null, canActPlayer bool not null, primary key (name), CONSTRAINT hasCharacter FOREIGN KEY(name) REFERENCES users(name), CONSTRAINT isEntity FOREIGN KEY(entity) REFERENCES entity01(entity));
create table enemies01 (id SERIAL, entity int not null, name varchar(255) not null, nameEnemy varchar(255) not null, stateEnemy varchar(255) not null, CONSTRAINT hasCharacter FOREIGN KEY(name) REFERENCES users(name), CONSTRAINT isEntity FOREIGN KEY(entity) REFERENCES entity01(entity));

\q para cerrar
\dt show tables
